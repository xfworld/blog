#### 一、AI嵌入模型概述  
  
##### 1.1 向量嵌入技术原理  
  
向量嵌入（Vector Embedding） 是将文本、图像、音频等非结构化数据映射到高维向量空间的技术，使得语义相似的数据在向量空间中距离更近。其数学本质是一个编码函数：  
  
$$\mathcal{E}: \mathcal{X} \rightarrow \mathbb{R}^d$$  
  
其中 $\mathcal{X}$ 是输入空间（如文本），$d$ 是嵌入维度（如768、1536）。  
  
嵌入模型的核心特性：  
  
| 特性 | 说明 | 技术影响 |  
|-----|------|---------|  
| 语义保持 | 相似输入产生相近向量 | 决定检索准确率 |  
| 维度选择 | 常见256-4096维 | 影响存储与计算成本 |  
| 归一化 | 输出向量通常单位化 | 便于余弦相似度计算 |  
| 上下文感知 | 如BERT考虑词序 | 提升语义理解能力 |  
  
> 行业数据：根据斯坦福大学《AI Index Report 2025》，92%的企业级AI应用使用向量嵌入技术，其中78%采用PostgreSQL+pgvector作为向量存储方案。

方案一：psycopg + 原生SQL（高性能）  
  
```python  
import psycopg  
from openai import OpenAI  
import numpy as np  
  
# 初始化客户端  
client = OpenAI(api_key="sk-...")  
conn = psycopg.connect("postgresql://user:pass@localhost:5432/ai_db")  
  
def generate_embedding(text: str, model: str = "text-embedding-3-small") -> list:  
    """调用OpenAI API生成嵌入"""  
    response = client.embeddings.create(input=text, model=model)  
    return response.data[0].embedding  
  
def insert_document(conn, doc_id: int, text: str, metadata: dict):  
    """生成嵌入并插入数据库"""  
    embedding = generate_embedding(text)  
      
    with conn.cursor() as cur:  
        cur.execute("""  
            INSERT INTO documents (doc_id, content, embedding, metadata, created_at)  
            VALUES (%s, %s, %s, %s, NOW())  
            ON CONFLICT (doc_id) DO UPDATE   
            SET embedding = EXCLUDED.embedding,  
                content = EXCLUDED.content,  
                metadata = EXCLUDED.metadata,  
                updated_at = NOW()  
        """, (doc_id, text, embedding, json.dumps(metadata)))  
    conn.commit()  
```

##### 2.2 批量导入优化策略  
  
策略一：批量生成 + 批量插入  
  
```python  
def batch_embed_and_insert(texts: list[str], batch_size: int = 100):  
    """批量处理：减少API调用开销 + 数据库事务优化"""  
    conn = psycopg.connect("postgresql://...")  
      
    for i in range(0, len(texts), batch_size):  
        batch = texts[i:i+batch_size]  
          
        # 批量调用嵌入API（注意速率限制）  
        response = client.embeddings.create(  
            input=batch,   
            model="text-embedding-3-small"  
        )  
        embeddings = [e.embedding for e in response.data]  
          
        # 批量插入数据库  
        with conn.cursor() as cur:  
            psycopg.extras.execute_batch(  
                cur,  
                """  
                INSERT INTO documents (content, embedding, created_at)  
                VALUES (%s, %s, NOW())  
                """,  
                list(zip(batch, embeddings))  
            )  
        conn.commit()  
```  
  
策略二：异步流水线处理  
  
```python  
import asyncio  
from concurrent.futures import ThreadPoolExecutor  
  
async def async_embed_pipeline(texts: list[str], max_concurrent: int = 20):  
    """异步嵌入生成 + 数据库写入流水线"""  
      
    async def embed_single(text: str) -> tuple[str, list]:  
        loop = asyncio.get_event_loop()  
        # 在线程池中执行阻塞的API调用  
        embedding = await loop.run_in_executor(  
            None,   
            lambda: generate_embedding(text)  
        )  
        return text, embedding  
      
    # 控制并发数  
    semaphore = asyncio.Semaphore(max_concurrent)  
      
    async def bounded_embed(text: str):  
        async with semaphore:  
            return await embed_single(text)  
      
    # 执行批量任务  
    tasks = [bounded_embed(text) for text in texts]  
    results = await asyncio.gather(*tasks)  
      
    # 批量写入数据库  
    conn = await asyncpg.connect("postgresql://...")  
    await conn.executemany(  
        """  
        INSERT INTO documents (content, embedding) VALUES ($1, $2)  
        """,  
        results  
    )  
    await conn.close()  
```

##### 3.2 PostgreSQL实现混合检索  
  
```sql  
-- 1. 创建支持混合检索的表结构  
CREATE TABLE knowledge_docs (  
    doc_id BIGINT PRIMARY KEY,  
    title VARCHAR(500),  
    content TEXT,  
    -- 向量嵌入  
    embedding vector(1536),  
    -- 关键词索引（GIN + tsvector）  
    content_tsv tsvector,  
    -- 结构化元数据  
    metadata JSONB,  
    -- 访问控制  
    acl_roles TEXT[],  -- 允许访问的角色列表  
    created_at TIMESTAMPTZ DEFAULT NOW(),  
    updated_at TIMESTAMPTZ DEFAULT NOW()  
);  
  
-- 2. 创建多类型索引  
-- 向量索引（余弦相似度）  
CREATE INDEX idx_doc_embedding ON knowledge_docs   
USING hnsw (embedding vector_cosine_ops)  
WITH (m = 16, ef_construction = 64);  
  
-- 全文搜索索引  
CREATE INDEX idx_doc_content ON knowledge_docs   
USING GIN (content_tsv);  
  
-- 元数据索引（常用过滤字段）  
CREATE INDEX idx_doc_category ON knowledge_docs   
USING BTREE ((metadata->>'category'));  
CREATE INDEX idx_doc_created ON knowledge_docs   
USING BTREE (created_at);  
  
-- 3. 混合检索函数（向量+关键词+元数据）  
CREATE OR REPLACE FUNCTION hybrid_search(  
    query_vector vector(1536),  
    query_text TEXT,  
    filter_json JSONB DEFAULT NULL,  
    k INT DEFAULT 10,  
    alpha FLOAT DEFAULT 0.7  -- 向量权重(0-1)  
)  
RETURNS TABLE (  
    doc_id BIGINT,  
    title VARCHAR,  
    content TEXT,  
    vector_score FLOAT,  
    keyword_score FLOAT,  
    combined_score FLOAT,  
    metadata JSONB  
) AS $$  
DECLARE  
    query_tsv tsvector := to_tsquery('english', query_text);  
BEGIN  
    RETURN QUERY  
    WITH vector_results AS (  
        SELECT   
            doc_id,  
            1 - (embedding <=> query_vector) AS v_score  
        FROM knowledge_docs  
        WHERE   
            (filter_json IS NULL OR metadata @> filter_json)  
            AND (acl_roles IS NULL OR current_user = ANY(acl_roles))  
        ORDER BY embedding <=> query_vector  
        LIMIT k * 2  -- 扩大候选集  
    ),  
    keyword_results AS (  
        SELECT   
            doc_id,  
            ts_rank(content_tsv, query_tsv) AS k_score  
        FROM knowledge_docs  
        WHERE   
            content_tsv @@ query_tsv  
            AND (filter_json IS NULL OR metadata @> filter_json)  
            AND (acl_roles IS NULL OR current_user = ANY(acl_roles))  
        ORDER BY k_score DESC  
        LIMIT k * 2  
    ),  
    -- 倒数融合（Reciprocal Rank Fusion）  
    fused AS (  
        SELECT   
            COALESCE(v.doc_id, k.doc_id) AS doc_id,  
            COALESCE(v.v_score, 0) AS vector_score,  
            COALESCE(k.k_score, 0) AS keyword_score,  
            -- RRF公式: Σ(1/(rank + 60))  
            (COALESCE(60.0 / NULLIF(  
                (SELECT COUNT(*)+1 FROM vector_results v2 WHERE v2.v_score > v.v_score), 0), 0) +  
             COALESCE(60.0 / NULLIF(  
                (SELECT COUNT(*)+1 FROM keyword_results k2 WHERE k2.k_score > k.k_score), 0), 0)  
            ) AS rrf_score  
        FROM vector_results v  
        FULL OUTER JOIN keyword_results k ON v.doc_id = k.doc_id  
    )  
    SELECT   
        d.doc_id, d.title, d.content,  
        f.vector_score, f.keyword_score,  
        -- 加权融合：alpha*向量 + (1-alpha)*关键词  
        (alpha * f.vector_score + (1-alpha) * f.keyword_score) AS combined_score,  
        d.metadata  
    FROM fused f  
    JOIN knowledge_docs d ON f.doc_id = d.doc_id  
    ORDER BY combined_score DESC  
    LIMIT k;  
END;  
$$ LANGUAGE plpgsql;  
```

##### 3.3 上下文优化策略  
  
策略一：动态窗口截取  
  
```sql  
-- 创建内容分块表（支持长文档）  
CREATE TABLE doc_chunks (  
    chunk_id BIGINT PRIMARY KEY,  
    doc_id BIGINT REFERENCES documents(doc_id),  
    chunk_index INT,  -- 块序号  
    content TEXT,     -- 分块内容  
    embedding vector(1536),  
    -- 上下文指针  
    prev_chunk_id BIGINT,  
    next_chunk_id BIGINT,  
    -- 元数据继承  
    metadata JSONB  
);  
  
-- 检索时获取相邻块作为上下文  
CREATE OR REPLACE FUNCTION get_chunk_with_context(  
    target_chunk_id BIGINT,  
    context_blocks INT DEFAULT 1  
)  
RETURNS TABLE (  
    doc_id BIGINT,  
    full_content TEXT,  
    relevance_score FLOAT  
) AS $$  
BEGIN  
    RETURN QUERY  
    WITH RECURSIVE context_chain AS (  
        -- 锚点：目标块  
        SELECT   
            chunk_id, doc_id, content,   
            prev_chunk_id, next_chunk_id,  
            0 AS depth,  
            content AS accumulated  
        FROM doc_chunks WHERE chunk_id = target_chunk_id  
          
        UNION ALL  
          
        -- 向前扩展  
        SELECT   
            c.chunk_id, c.doc_id, c.content,  
            c.prev_chunk_id, c.next_chunk_id,  
            cc.depth + 1,  
            c.content || '\n\n' || cc.accumulated  
        FROM doc_chunks c  
        JOIN context_chain cc ON c.chunk_id = cc.prev_chunk_id  
        WHERE cc.depth < context_blocks  
          
        UNION ALL  
          
        -- 向后扩展  
        SELECT   
            c.chunk_id, c.doc_id, c.content,  
            c.prev_chunk_id, c.next_chunk_id,  
            cc.depth + 1,  
            cc.accumulated || '\n\n' || c.content  
        FROM doc_chunks c  
        JOIN context_chain cc ON c.chunk_id = cc.next_chunk_id  
        WHERE cc.depth < context_blocks  
    )  
    SELECT   
        doc_id,  
        STRING_AGG(content, '\n\n' ORDER BY chunk_index) AS full_content,  
        1.0 AS relevance_score  -- 实际可加入向量相似度  
    FROM context_chain  
    GROUP BY doc_id;  
END;  
$$ LANGUAGE plpgsql;  
```

##### 4.3 动态模型选择函数  
  
```sql  
CREATE OR REPLACE FUNCTION smart_retrieve(  
    query_text TEXT,  
    query_type TEXT DEFAULT 'general',  -- general/code/multilingual  
    k INT DEFAULT 10  
)  
RETURNS TABLE (  
    doc_id BIGINT,  
    content TEXT,  
    score FLOAT,  
    model_used TEXT,  
    metadata JSONB  
) AS $$  
DECLARE  
    query_bge vector(1024);  
    query_openai vector(1536);  
    query_code vector(768);  
BEGIN  
    -- 根据查询类型选择嵌入模型（实际应用中应在应用层生成）  
    -- 此处简化：假设已传入对应向量  
      
    IF query_type = 'code' THEN  
        -- 代码搜索：使用CodeBERT嵌入  
        RETURN QUERY  
        SELECT   
            doc_id, content,  
            1 - (embedding_code <=> query_code) AS score,  
            'codebert' AS model_used,  
            metadata  
        FROM multi_model_docs  
        WHERE metadata->>'type' = 'code'  
        ORDER BY embedding_code <=> query_code  
        LIMIT k;  
          
    ELSIF query_type = 'multilingual' THEN  
        -- 多语言：使用BGE-m3  
        RETURN QUERY  
        SELECT   
            doc_id, content,  
            1 - (embedding_bge <=> query_bge) AS score,  
            'bge-m3' AS model_used,  
            metadata  
        FROM multi_model_docs  
        ORDER BY embedding_bge <=> query_bge  
        LIMIT k;  
          
    ELSE  
        -- 通用场景：融合多模型结果  
        RETURN QUERY  
        WITH bge_results AS (  
            SELECT doc_id, 1 - (embedding_bge <=> query_bge) AS score  
            FROM multi_model_docs  
            ORDER BY embedding_bge <=> query_bge  
            LIMIT k*2  
        ),  
        openai_results AS (  
            SELECT doc_id, 1 - (embedding_openai <=> query_openai) AS score  
            FROM multi_model_docs  
            ORDER BY embedding_openai <=> query_openai  
            LIMIT k*2  
        )  
        SELECT   
            COALESCE(b.doc_id, o.doc_id),  
            d.content,  
            -- 简单平均融合  
            (COALESCE(b.score, 0) + COALESCE(o.score, 0)) / 2.0 AS score,  
            'fusion(bge+openai)' AS model_used,  
            d.metadata  
        FROM bge_results b  
        FULL OUTER JOIN openai_results o ON b.doc_id = o.doc_id  
        JOIN multi_model_docs d ON COALESCE(b.doc_id, o.doc_id) = d.doc_id  
        ORDER BY score DESC  
        LIMIT k;  
    END IF;  
END;  
$$ LANGUAGE plpgsql;  
```

##### 5.1 嵌入缓存策略  
  
应用层缓存（Redis）：  
  
```python  
import redis  
import hashlib  
import json  
  
class EmbeddingCache:  
    def __init__(self, redis_url: str, ttl: int = 86400):  
        self.redis = redis.from_url(redis_url)  
        self.ttl = ttl  
      
    def _cache_key(self, text: str, model: str) -> str:  
        """生成缓存key：hash(text+model)"""  
        content = f"{text}::{model}"  
        return f"embed:{hashlib.sha256(content.encode()).hexdigest()[:16]}"  
      
    def get(self, text: str, model: str) -> list | None:  
        """获取缓存的嵌入"""  
        key = self._cache_key(text, model)  
        cached = self.redis.get(key)  
        if cached:  
            return json.loads(cached)  
        return None  
      
    def set(self, text: str, model: str, embedding: list):  
        """缓存新生成的嵌入"""  
        key = self._cache_key(text, model)  
        self.redis.setex(key, self.ttl, json.dumps(embedding))  
  
# 使用示例  
cache = EmbeddingCache("redis://localhost:6379")  
  
def cached_generate_embedding(text: str, model: str = "text-embedding-3-small"):  
    # 1. 尝试缓存  
    cached = cache.get(text, model)  
    if cached:  
        return cached  
      
    # 2. 调用API生成  
    embedding = generate_embedding(text, model)  
      
    # 3. 写入缓存  
    cache.set(text, model, embedding)  
      
    return embedding  
```

WITH (m = 16, ef_construction = 64);  
  
CREATE INDEX idx_faq_category ON faq_knowledge (category);  
  
-- 3. 同义词扩展检索函数  
CREATE OR REPLACE FUNCTION search_faq(  
    user_query TEXT,  
    category_filter TEXT DEFAULT NULL,  
    k INT DEFAULT 5  
)  
RETURNS TABLE (  
    faq_id BIGINT,  
    question TEXT,  
    answer TEXT,  
    similarity FLOAT,  
    matched_synonym TEXT  
) AS $$  
DECLARE  
    query_vec vector(1024);  
BEGIN  
    -- 应用层生成查询向量（此处简化）  
    query_vec := $1::vector(1024);  
      
    RETURN QUERY  
    SELECT   
        f.faq_id, f.question, f.answer,  
        1 - (f.embedding <=> query_vec) AS similarity,  
        -- 记录匹配的同义词  
        (SELECT s FROM unnest(f.synonyms) s   
         WHERE to_tsvector('simple', s) @@ to_tsquery('simple', user_query)  
         LIMIT 1) AS matched_synonym  
    FROM faq_knowledge f  
    WHERE   
        (category_filter IS NULL OR f.category = category_filter)  
        AND f.metadata->>'status' = 'active'  -- 仅检索生效条目  
    ORDER BY f.embedding <=> query_vec  
    LIMIT k;  
END;  
$$ LANGUAGE plpgsql;

CREATE OR REPLACE FUNCTION search_code_by_natural_language(  
    nl_query TEXT,  
    language_filter TEXT DEFAULT NULL,  
    k INT DEFAULT 10  
)  
RETURNS TABLE (  
    snippet_id BIGINT,  
    repo_name VARCHAR,  
    file_path TEXT,  
    code_preview TEXT,  -- 截取前200字符  
    doc_similarity FLOAT,  
    metadata JSONB  
) AS $$

    query_vec vector(768);  
BEGIN  
    -- 应用层：用BGE模型将NL查询转为向量  
    query_vec := $1::vector(768);

    SELECT   
        cs.snippet_id, cs.repo_name, cs.file_path,  
        LEFT(cs.code_content, 200) || '...' AS code_preview,  
        1 - (cs.doc_embedding <=> query_vec) AS doc_similarity,

            'function', cs.function_name,  
            'tags', cs.tags,  
            'complexity', cs.complexity_score  
        ) AS metadata  
    FROM code_snippets cs  
    WHERE   
        (language_filter IS NULL OR cs.language = language_filter)  
        AND cs.doc_embedding IS NOT NULL  
    ORDER BY cs.doc_embedding <=> query_vec  
    LIMIT k;  
END;  
$$ LANGUAGE plpgsql;

##### 6.3 案例三：多模态电商检索  
  
场景：电商平台支持"以图搜图"+"文本描述"混合检索商品。  
  
技术方案：  
  
```sql  
-- 1. 商品多模态表  
CREATE TABLE products_multimodal (  
    product_id BIGINT PRIMARY KEY,  
    title VARCHAR(300),  
    description TEXT,  
    category_path VARCHAR(100),  -- 如"家电/厨房/咖啡机"  
      
    -- 文本嵌入（商品描述语义）  
    text_embedding vector(768),  
      
    -- 图像嵌入（CLIP视觉特征）  
    image_embedding halfvec(512),  -- 半精度节省存储  
      
    -- 商品属性（结构化过滤）  
    attributes JSONB,  -- {brand: "Delonghi", price_range: "1000-2000", ...}  
      
    -- 业务字段  
    price DECIMAL(10,2),  
    stock INT,  
    created_at TIMESTAMPTZ  
);  
  
-- 2. 多索引策略  
CREATE INDEX idx_prod_text ON products_multimodal   
USING hnsw (text_embedding vector_cosine_ops);  
  
CREATE INDEX idx_prod_image ON products_multimodal   
USING hnsw (image_embedding halfvec_cosine_ops);  
  
CREATE INDEX idx_prod_category ON products_multimodal (category_path);  
CREATE INDEX idx_prod_price ON products_multimodal (price);  
  
-- 3. 多模态融合检索  
CREATE OR REPLACE FUNCTION multimodal_product_search(  
    text_query TEXT DEFAULT NULL,  
    image_query_vector halfvec(512) DEFAULT NULL,  
    price_min DECIMAL DEFAULT NULL,  
    price_max DECIMAL DEFAULT NULL,  
    category_filter TEXT DEFAULT NULL,  
    k INT DEFAULT 20,  
    text_weight FLOAT DEFAULT 0.6,  
    image_weight FLOAT DEFAULT 0.4  
)  
RETURNS TABLE (  
    product_id BIGINT,  
    title VARCHAR,  
    price DECIMAL,  
    text_score FLOAT,  
    image_score FLOAT,  
    final_score FLOAT,  
    thumbnail_url TEXT  
) AS $$  
DECLARE  
    text_vec vector(768);  
BEGIN  
    -- 应用层生成text_query的嵌入  
    text_vec := COALESCE($1::vector(768), '0'::vector(768));  
      
    RETURN QUERY  
    WITH text_matches AS (  
        SELECT   
            product_id,  
            CASE WHEN $1 IS NOT NULL   
                 THEN 1 - (text_embedding <=> text_vec)  
                 ELSE 0 END AS t_score  
        FROM products_multimodal  
        WHERE $1 IS NULL OR text_embedding IS NOT NULL  
        ORDER BY CASE WHEN $1 IS NOT NULL   
                      THEN text_embedding <=> text_vec END  
        LIMIT CASE WHEN $1 IS NOT NULL THEN k*2 ELSE 0 END  
    ),  
    image_matches AS (  
        SELECT   
            product_id,  
            CASE WHEN $2 IS NOT NULL   
                 THEN 1 - (image_embedding <=> $2)  
                 ELSE 0 END AS i_score  
        FROM products_multimodal  
        WHERE $2 IS NULL OR image_embedding IS NOT NULL  
        ORDER BY CASE WHEN $2 IS NOT NULL   
                      THEN image_embedding <=> $2 END  
        LIMIT CASE WHEN $2 IS NOT NULL THEN k*2 ELSE 0 END  
    ),  
    fused AS (  
        SELECT   
            COALESCE(t.product_id, i.product_id) AS pid,  
            COALESCE(t.t_score, 0) AS ts,  
            COALESCE(i.i_score, 0) AS is  
        FROM text_matches t  
        FULL OUTER JOIN image_matches i ON t.product_id = i.product_id  
    )  
    SELECT   
        p.product_id, p.title, p.price,  
        f.ts AS text_score,  
        f.is AS image_score,  
        -- 加权融合  
        ($7 * f.ts + $8 * f.is) AS final_score,  
        -- 生成缩略图URL（应用层处理）  
        CONCAT('/images/', p.product_id, '_thumb.jpg') AS thumbnail_url  
    FROM fused f  
    JOIN products_multimodal p ON f.pid = p.product_id  
    WHERE   
        ($3 IS NULL OR p.price >= $3)  
        AND ($4 IS NULL OR p.price <= $4)  
        AND ($5 IS NULL OR p.category_path LIKE $5 || '%')  
        AND p.stock > 0  -- 仅检索有库存商品  
    ORDER BY final_score DESC  
    LIMIT $6;  
END;  
$$ LANGUAGE plpgsql;  
```
